How to Enable Save-to-Sheet

1. Open your Google Sheet → Extensions → Apps Script
2. Paste this script, replace YOUR_SHEET_ID, click Save
3. Click Deploy → New Deployment → Web App
4. Set "Execute as: Me" and "Who has access: Anyone" → Deploy
5. Copy the Web App URL and paste it above 

function doPost(e) {
  var sheet = SpreadsheetApp
    .openById('YOUR_SHEET_ID')
    .getSheetByName('POT');
  var d = JSON.parse(e.postData.contents);
  sheet.appendRow([
    '', d.ministry, d.websiteSection, '', d.sitting,
    d.type, 'Ready to Vet', d.question, d.wordCount,
    '', d.comments, '', d.research, '', ''
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet() {
  return ContentService.createTextOutput('PQ Drafter Active');
}
